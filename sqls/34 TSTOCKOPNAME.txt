CREATE TABLE TSTOCKOPNAME(
	ID INT PRIMARY KEY IDENTITY(1,1),
	REFNO VARCHAR(30) UNIQUE,
	TRANSDATE DATE,	
	WAREHOUSE_ID INT FOREIGN KEY REFERENCES TWAREHOUSE(ID),
	NOTES VARCHAR(200),	
	CLOSED SMALLINT,
	TRANSTYPE SMALLINT,
	MODIFIEDDATE DATETIME2(0),
	MODIFIEDBY VARCHAR(30)
)

CREATE TABLE TSTOCKOPNAMEITEM(
	ID INT PRIMARY KEY IDENTITY(1,1),
	STOCKOPNAME_ID INT,
	TRANSDATE DATE,
	TRANSTYPE SMALLINT,
	ITEM_ID INT,
	UOM_ID INT,
	QTY FLOAT NOT NULL,
	QTYSYS FLOAT NOT NULL,
	VARIANT FLOAT NOT NULL,
	KONVERSI FLOAT NOT NULL,
	WAREHOUSE_ID INT FOREIGN KEY REFERENCES TWAREHOUSE(ID),	
	HARGAAVG FLOAT NOT NULL,
	LASTCOST FLOAT NOT NULL	
)

GO
CREATE TABLE TSTOCKADJUSTMENT(
	ID INT PRIMARY KEY IDENTITY(1,1),
	REFNO VARCHAR(30) UNIQUE,
	TRANSDATE DATE,	
	STOCKOPNAME_ID INT FOREIGN KEY REFERENCES TSTOCKOPNAME(ID),
	WAREHOUSE_ID INT FOREIGN KEY REFERENCES TWAREHOUSE(ID),
	NOTES VARCHAR(200),	
	CLOSED SMALLINT,
	TRANSTYPE SMALLINT,
	MODIFIEDDATE DATETIME2(0),
	MODIFIEDBY VARCHAR(30)
)
