CREATE TABLE TFINANCIALTRANSACTION(
	ID INT PRIMARY KEY IDENTITY(1,1),
	REFNO VARCHAR(100),
	TRANSDATE DATE,
	HEADER_ID INT NOT NULL,
	HEADER_FLAG INT NOT NULL,
	TRANSTYPE SMALLINT,
	MEDIA SMALLINT,
	MEDIANO VARCHAR(100),
	REKENING_ID INT FOREIGN KEY REFERENCES TREKENING(ID),
	ACCOUNT_ID INT FOREIGN KEY REFERENCES TACCOUNT(ID),
	DEBETAMT FLOAT NOT NULL,
	CREDITAMT FLOAT NOT NULL,
	AMOUNT FLOAT NOT NULL,
	RETURAMT FLOAT NOT NULL,
	SALESINVOICE_ID INT FOREIGN KEY REFERENCES TSALESINVOICE(ID),
	SALESRETUR_ID INT FOREIGN KEY REFERENCES TSALESRETUR(ID),
	PURCHASEINVOICE_ID INT FOREIGN KEY REFERENCES TPURCHASEINVOICE(ID),
	PURCHASERETUR_ID INT FOREIGN KEY REFERENCES TPURCHASERETUR(ID),
	NOTES VARCHAR(300)
)
