CREATE TABLE TPURCHASEINVOICE(
	ID INT PRIMARY KEY IDENTITY(1,1),
	INVOICENO VARCHAR(30) UNIQUE,
	REFERENSI VARCHAR(30),
	TRANSDATE DATE,	
	SUPPLIER_ID INT,
	DUEDATE DATE,
	PAYMENTFLAG SMALLINT,
	NOTES VARCHAR(200),
	SUBTOTAL FLOAT,
	PPN FLOAT,
	AMOUNT FLOAT,	
	PAIDAMOUNT FLOAT,
	RETURAMOUNT FLOAT,
	WAREHOUSE_ID INT FOREIGN KEY REFERENCES TWAREHOUSE(ID),
	REKENING_ID INT FOREIGN KEY REFERENCES TREKENING(ID),
	STATUS SMALLINT,
	CLOSED SMALLINT,
	PAIDOFF SMALLINT,
	PAIDOFFDATE DATE,
	MODIFIEDDATE DATETIME2(0),
	MODIFIEDBY VARCHAR(30)
)
