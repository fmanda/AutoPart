CREATE TABLE TSALESRETUR(
	ID INT PRIMARY KEY IDENTITY(1,1),
	REFNO VARCHAR(30) UNIQUE,
	TRANSDATE DATE,	
	INVOICE_ID INT FOREIGN KEY REFERENCES TSALESINVOICE(ID),
	CUSTOMER_ID INT FOREIGN KEY REFERENCES TCUSTOMER(ID),
	SALESMAN_ID INT FOREIGN KEY REFERENCES TSALESMAN(ID),
	DUEDATE DATE,
	RETURFLAG SMALLINT,
	NOTES VARCHAR(200),
	SUBTOTAL FLOAT,
	PPN FLOAT,
	AMOUNT FLOAT,	
	PAIDAMOUNT FLOAT,
	WAREHOUSE_ID INT FOREIGN KEY REFERENCES TWAREHOUSE(ID),
	STATUS SMALLINT,
	CLOSED SMALLINT,
	MODIFIEDDATE DATETIME2(0),
	MODIFIEDBY VARCHAR(30)
)
