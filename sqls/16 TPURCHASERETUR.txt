CREATE TABLE TPURCHASERETUR(
	ID INT PRIMARY KEY IDENTITY(1,1),
	REFNO VARCHAR(30) UNIQUE,
	TRANSDATE DATE,	
	INVOICE_ID INT FOREIGN KEY REFERENCES TPURCHASEINVOICE(ID),
	SUPPLIER_ID INT,
	DUEDATE DATE,
	RETURFLAG SMALLINT,
	NOTES VARCHAR(200),
	SUBTOTAL FLOAT,
	PPN FLOAT,
	AMOUNT FLOAT,	
	PAIDAMOUNT FLOAT,
	RETURAMOUNT FLOAT,
	WAREHOUSE_ID INT,
	STATUS SMALLINT,
	CLOSED SMALLINT,
	MODIFIEDDATE DATETIME2(0),
	MODIFIEDBY VARCHAR(30)
)
