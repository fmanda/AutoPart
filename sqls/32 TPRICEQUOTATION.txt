CREATE TABLE TPRICEQUOTATION(
	ID INT PRIMARY KEY IDENTITY(1,1),
	REFNO VARCHAR(100),
	TRANSDATE DATE,		
	NOTES VARCHAR(300),
	ISACTIVE SMALLINT,
	MODIFIEDDATE DATETIME2(0),
	MODIFIEDBY VARCHAR(30)
)

CREATE TABLE TPRICEQUOTATIONITEM(
	ID INT PRIMARY KEY IDENTITY(1,1),
	QUOTATION_ID INT,
	ITEM_ID INT,
	UOM_ID INT,
	KONVERSI FLOAT,
	HARGABELI FLOAT,
	HARGAAVG FLOAT,
	HARGAJUAL1 FLOAT,
	HARGAJUAL2 FLOAT,
	HARGAJUAL3 FLOAT,
	HARGAJUAL4 FLOAT,
	MODIFIEDDATE DATETIME2(0),
	MODIFIEDBY VARCHAR(30)
)
