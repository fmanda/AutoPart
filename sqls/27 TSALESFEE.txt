CREATE TABLE TSETTINGFEE(
	ID INT PRIMARY KEY IDENTITY(1,1),
	NAMA VARCHAR(100),
	FEE_1 FLOAT,
	FEE_2 FLOAT,
	FEE_3 FLOAT,
	MAXTEMPO_1 INT,
	MAXTEMPO_2 INT,
	MAXTEMPO_3 INT,
	ISACTIVE SMALLINT
)

CREATE TABLE TSALESFEE(
	ID INT PRIMARY KEY IDENTITY(1,1),
	REFNO VARCHAR(30) UNIQUE,
	TRANSDATE DATE,
	SALESINVOICE_ID INT,
	SETTINGFEE_ID INT,
	SALESMAN_ID INT,
	SALESAMT FLOAT,
	RETURAMT FLOAT,
	SALESCOST FLOAT,
	RETURCOST FLOAT,
	NETPROFIT FLOAT,
	FEE FLOAT,
	PAIDOFFDATE DATE,
	STATUS SMALLINT
)
