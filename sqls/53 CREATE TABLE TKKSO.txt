CREATE TABLE TKKSO(
	ID INT PRIMARY KEY IDENTITY(1,1),
	REFNO VARCHAR(30) UNIQUE,
	TRANSDATE DATE,	
	WAREHOUSE_ID INT FOREIGN KEY REFERENCES TWAREHOUSE(ID),
	NOTES VARCHAR(200),	
	RAK VARCHAR(30),
	STOCKOPNAME_ID INT,
	PIC VARCHAR(30),	
	MODIFIEDDATE DATETIME2(0),
	MODIFIEDBY VARCHAR(30)
)

CREATE TABLE TKKSOITEM(
	ID INT PRIMARY KEY IDENTITY(1,1),
	KKSO_ID INT,
	ITEM_ID INT,
	UOM_ID INT,
	QTY FLOAT NOT NULL,	
	KONVERSI FLOAT
)

CREATE TABLE TSTOCKOPNAMEKKSO(
	ID INT PRIMARY KEY IDENTITY(1,1),
	KKSO_ID INT,
	STOCKOPNAME_ID INT
)
