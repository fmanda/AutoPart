CREATE TABLE TTRANSDETAIL(
	ID INT PRIMARY KEY IDENTITY(1,1),
	TRANSDATE DATE,
	HEADER_ID INT NOT NULL,
	HEADER_FLAG INT NOT NULL,
	TRANSTYPE SMALLINT,
	ITEM_ID INT FOREIGN KEY REFERENCES TITEM(ID),
	UOM_ID INT FOREIGN KEY REFERENCES TUOM(ID),
	QTY FLOAT NOT NULL,
	KONVERSI FLOAT NOT NULL,
	WAREHOUSE_ID INT FOREIGN KEY REFERENCES TWAREHOUSE(ID),
	HARGA FLOAT NOT NULL,
	DISCOUNT FLOAT NOT NULL,
	HARGAAVG FLOAT NOT NULL,
	LASTCOST FLOAT NOT NULL,
	PPN FLOAT,
	TOTAL FLOAT,
	PRICETYPE SMALLINT,
	REFNO VARCHAR(100),
	TRANSDEF VARCHAR(100),
)
