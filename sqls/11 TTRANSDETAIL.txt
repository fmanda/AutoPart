CREATE TABLE TTRANSDETAIL(
	ID INT PRIMARY KEY IDENTITY(1,1),
	TRANSDATE DATE,
	HEADER_ID INT,
	HEADER_FLAG INT,
	TRANSTYPE SMALLINT,
	ITEM_ID INT FOREIGN KEY REFERENCES TITEM(ID),
	UOM_ID INT FOREIGN KEY REFERENCES TUOM(ID),
	QTY FLOAT,
	KONVERSI FLOAT,
	WAREHOUSE_ID INT FOREIGN KEY REFERENCES TWAREHOUSE(ID),
	HARGA FLOAT,
	DISCOUNT FLOAT,
	HARGAAVG FLOAT,
	PPN FLOAT,
	TOTAL FLOAT,
	REFNO VARCHAR(100),
	TRANSDEF VARCHAR(100)
)
