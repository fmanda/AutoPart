CREATE TABLE TENDOFDAY(
	ID INT PRIMARY KEY IDENTITY(1,1),
	REFNO VARCHAR(30) UNIQUE,
	TRANSDATE DATE,	
	NOTES VARCHAR(200),
	MODIFIEDDATE DATETIME2(0),
	MODIFIEDBY VARCHAR(30)
)

CREATE TABLE TCASHOPNAME(
	ID INT PRIMARY KEY IDENTITY(1,1),
	REFNO VARCHAR(30) UNIQUE,
	TRANSDATE DATE,	
	NOTES VARCHAR(200),
	ENDOFDAY_ID INT,
	TOTALVARIANT FLOAT,
	MODIFIEDDATE DATETIME2(0),
	MODIFIEDBY VARCHAR(30)
)

CREATE TABLE TCASHOPNAMEITEM(
	ID INT PRIMARY KEY IDENTITY(1,1),
	CASHOPNAME_ID INT,
	REKENING_ID INT,
	SALDOSYSTEM FLOAT,
	SALDOFISIK FLOAT,
	VARIANT FLOAT,
	NOTES VARCHAR(200)
)
