CREATE TABLE TWAREHOUSE(
	ID INT PRIMARY KEY IDENTITY(1,1),
	KODE VARCHAR(30) UNIQUE,
	NAMA VARCHAR(100),		
	ISACTIVE SMALLINT,
	JENIS SMALLINT,
	IS_EXTERNAL SMALLINT DEFAULT 0,
	PROJECT_CODE VARCHAR(30)
)


CREATE TABLE TRACK(
	ID INT PRIMARY KEY IDENTITY(1,1),
	WAREHOUSE_ID INT,
	RAK VARCHAR(100),
	LOKASI VARCHAR(100)
);
