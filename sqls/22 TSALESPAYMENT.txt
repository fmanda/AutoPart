CREATE TABLE TSALESPAYMENT(
	ID INT PRIMARY KEY IDENTITY(1,1),
	REFNO VARCHAR(100),
	TRANSDATE DATE,	
	DUEDATE DATE,	
	PAYMENTFLAG SMALLINT,
	SALESMAN_ID INT,
	MEDIA SMALLINT,
	MEDIANO VARCHAR(100),
	REKENING_ID INT FOREIGN KEY REFERENCES TREKENING(ID),
	AMOUNT FLOAT,
	RETURAMOUNT FLOAT,	
	NOTES VARCHAR(300),
	CLOSED SMALLINT,
	MODIFIEDDATE DATETIME2(0),
	MODIFIEDBY VARCHAR(30)
)
