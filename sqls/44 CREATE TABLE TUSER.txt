CREATE TABLE TTASK(
	ID INT PRIMARY KEY IDENTITY(1,1),
	TASKCODE VARCHAR(50),
	TASKNAME VARCHAR(100),
	GROUPNAME VARCHAR(100)
);

CREATE TABLE TUSER(
	ID INT PRIMARY KEY IDENTITY(1,1),
	USERNAME VARCHAR(30),
	PASSWORD VARCHAR(30),
	NAMA VARCHAR(100)
);

CREATE TABLE TUSERTASK(
	ID INT PRIMARY KEY IDENTITY(1,1),
	USER_ID INT,
	TASK_ID INT,
	DOACCESS SMALLINT,
	DOCREATE SMALLINT,
	DOEDIT SMALLINT,
	DODELETE SMALLINT
);
