CREATE TABLE TCUSTOMER(
	ID INT PRIMARY KEY IDENTITY(1,1),
	KODE VARCHAR(30) UNIQUE,
	NAMA VARCHAR(100),	
	ALAMAT VARCHAR(200),	
	TELP VARCHAR(30),
	ISACTIVE SMALLINT,
	TEMPO INT,
	TIPEHARGA SMALLINT,
	MODIFIEDDATE DATETIME2(0),
	MODIFIEDBY VARCHAR(30)
)

CREATE TABLE TSUPPLIER(
	ID INT PRIMARY KEY IDENTITY(1,1),
	KODE VARCHAR(30) UNIQUE,
	NAMA VARCHAR(100),	
	ALAMAT VARCHAR(200),	
	TELP VARCHAR(30),
	ISACTIVE SMALLINT,
	TEMPO INT,
	MODIFIEDDATE DATETIME2(0),
	MODIFIEDBY VARCHAR(30)
)
