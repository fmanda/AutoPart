CREATE TABLE TREKENING(
	ID INT PRIMARY KEY IDENTITY(1,1),
	KODE VARCHAR(30) UNIQUE,
	NAMA VARCHAR(100),		
	ISACTIVE SMALLINT,
	JENIS SMALLINT
)


CREATE TABLE TACCOUNT(
	ID INT PRIMARY KEY IDENTITY(1,1),
	KODE VARCHAR(30) UNIQUE,
	NAMA VARCHAR(100),		
	ISACTIVE SMALLINT,
	PARENT_ID SMALLINT,
	JENIS SMALLINT,
	ISCREDIT SMALLINT,
	ISDETAIL SMALLINT
)
