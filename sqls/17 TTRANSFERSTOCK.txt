CREATE TABLE TTRANSFERSTOCK(
	ID INT PRIMARY KEY IDENTITY(1,1),
	REFNO VARCHAR(30) UNIQUE,
	TRANSDATE DATE,	
	WH_ASAL_ID INT FOREIGN KEY REFERENCES TWAREHOUSE(ID),
	WH_TUJUAN_ID INT FOREIGN KEY REFERENCES TWAREHOUSE(ID),	
	NOTES VARCHAR(200),	
	CLOSED SMALLINT,
	MODIFIEDDATE DATETIME2(0),
	MODIFIEDBY VARCHAR(30)
)
