CREATE TABLE TSALESINVOICE(
	ID INT PRIMARY KEY IDENTITY(1,1),
	INVOICENO VARCHAR(30) UNIQUE,
	TRANSDATE DATE,		
	DUEDATE DATE,
	PAYMENTFLAG SMALLINT,
	CUSTOMER_ID INT FOREIGN KEY REFERENCES TCUSTOMER(ID),
	SALESMAN_ID INT FOREIGN KEY REFERENCES TSALESMAN(ID),
	MEKANIK_ID INT FOREIGN KEY REFERENCES TMEKANIK(ID),
	SALESTYPE SMALLINT,
	NOTES VARCHAR(200),
	SUBTOTAL FLOAT,
	PPN FLOAT,
	AMOUNT FLOAT,	
	PAIDAMOUNT FLOAT,
	RETURAMOUNT FLOAT,
	WAREHOUSE_ID INT FOREIGN KEY REFERENCES TWAREHOUSE(ID),
	REKENING_ID INT FOREIGN KEY REFERENCES TREKENING(ID),
	SETTINGFEE_ID INT,
	STATUS SMALLINT,
	CLOSED SMALLINT,
	PAIDOFF SMALLINT,
	PAIDOFFDATE DATE,
	CASHAMOUNT FLOAT,
	MODIFIEDDATE DATETIME2(0),
	MODIFIEDBY VARCHAR(30)
)
